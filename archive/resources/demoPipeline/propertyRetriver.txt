rm -rf conf 
mkdir conf
mkdir -p conf/activemqInt

mkdir -p conf/activemqExt
mkdir -p conf/tomcat7


ssh -t tomcat7@$demo_server "find /usr/local/tomcat/default/conf/ -type f -name '*.properties'" >> ${WORKSPACE}/tomcat7-file_list.txt
ssh -t tomcat7@$demo_server "find /opt/apache-activemq-5.14.0/ELIS2-int/conf/ -type f -name 'activemq.xml'" >> ${WORKSPACE}/activemqInt-file_list.txt
ssh -t tomcat7@$demo_server "find /opt/apache-activemq-5.14.0/ELIS2-ext/conf/ -type f -name 'activemq.xml'" >> ${WORKSPACE}/activemqExt-file_list.txt

while read filename; do
        scp tomcat7@$demo_server:$filename ${WORKSPACE}/conf/tomcat7;
done < ${WORKSPACE}/tomcat7-file_list.txt
cd conf

while read activeInt; do
        scp tomcat7@$demo_server:$activeInt ${WORKSPACE}/conf/activemqInt;
done < ${WORKSPACE}/activemqInt-file_list.txt

while read activeExt; do
        scp tomcat7@$demo_server:$activeExt ${WORKSPACE}/conf/activemqExt;
done < ${WORKSPACE}/activemqExt-file_list.txt

ls -lhart 
cd ..
zip -r conf-${BUILD_NUMBER}-${demo_server}_properties.zip conf
unzip -l conf-${BUILD_NUMBER}-${demo_server}_properties.zip


ls -lhart ${WORKSPACE}/conf-${BUILD_NUMBER}-${demo_server}_properties.zip
